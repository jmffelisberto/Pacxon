 public class Stats {

         this.score = 0;
         this.progressPercentage = 0;
     }

         return score;
     }

     }

         return progressPercentage;
     }
 
     public void addScore(int points) {
         this.score += points;
     }

         this.progressPercentage = (int)(((double) filledBlocks / totalBlocks) * 100);
     }

     }

     }
